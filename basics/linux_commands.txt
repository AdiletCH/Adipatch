ls = показывает содержимое папки (файлы и другие папки)
ls название_директории{py33 }= показывает все файлы и папки внутри папки py33 
ls -a = показывает скрытые файлы и папки
ls -l = показывает дополнительныую информацию о файлах и директориях
ls -la = показывает скрытые файлы и директорию с подробной информациой

cd = переход домашнюю директорию
cd ~ =  переход в домашнюю директорию
cd test = переход в папку test,  если такой директории нет, то выйдет ошибка
cd .. = переход на директорию выше 
cd ../.. =  переход на две директорию 
cd - = cntr + z

mkdir test - создает папку  test в текущей директории
mkdir test/test2 = создает папку test2 внутри папки test, если папка существует
mkdir test test2 = создает папку test и test2 в текущей директории

touch file.txt = создает файлы 'file' c разрешением 'txt' 
touch file.txt file.txt = создает файл 'file1.txt' и file2.txt  

rm test.txt = удаляет файл
rmdir test = удаляет пустую папку  test
rm -rf test удаляет всё; и файлы и директории с содержимым и пустые директории. Рекурсивное удаление (полное)

mv test.txt test = перемещает файл test.txt в директорию test
mv test.txt test2.txt = переименовывает файл test1.txt на test2.txt

pwd (print working directory)= показывет аблсолютный путь (начиная с корневой папки /) до текущей директории

history = показывает все команды, которые мы вводили данный терминал
history 10 = показывает 10 последних введеных команд из истории команд
history | grep test = показывает кооманды из истории команд содержавшие слово test

nano test.txt = открывает файл test.txt в текстовом редакторе nano

cat test.txt = выводит содержимое файла test.txt в терминал

sudo(super user do) = право суперюзера

apt = системный установщик пакетов (ubuntu)
brew = установщик пакетов (macos)
